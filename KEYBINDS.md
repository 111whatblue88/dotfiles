#i3

alt+z = rofi
alt+x = alacritty 
alt+c = firefox 

alt+1-9 = workspace 1-9
alt+shift+1-9 = move window to workspace 1-9

alt+h = move left
alt+j = move down
alt+k = move up
alt+l = move right
to move windows, its the same except its alt+shift+*,

alt+d = kill
alt+f = fullscreen

alt+s = stacked layout
alt+w = tabbed layout
alt+e = pane layout

alt+t = split horizontally
alt+r = split vertically

alt+shift+space = toggle floating
alt+shift+s = screenshot

alt+shift+i = mute audio
alt+shift+o = audio volume -10
alt+shift+p = audio volume +10

alt+shift+r = reload i3
alt+shift+a = log out

#Alacritty/Terminal

alt+g = autocomplete accept

alt+- = decrease font
alt+= = increase font
